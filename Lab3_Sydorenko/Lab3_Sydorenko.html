import random
import time
import matplotlib.pyplot as plt

def generate_random_list(size, lower_bound, upper_bound):

    return [random.randint(lower_bound, upper_bound) for _ in range(size)]

def bubble_sort(arr):
    n = len(arr)
    for i in range(n):
        swapped = False
        for j in range(0, n-i-1):
            if arr[j] > arr[j+1]:
                arr[j], arr[j+1] = arr[j+1], arr[j]
                swapped = True
        if not swapped:
            break
    return arr

def insertion_sort(nums):
    for j in range(1, len(nums)):
        key = nums[j]
        i = j - 1
        while i >= 0 and nums[i] > key:
            nums[i + 1] = nums[i]
            i -= 1
        nums[i + 1] = key
    return nums

sizes = [5, 10, 50, 100, 500, 1000, 2000, 3000, 4000, 5000]
times_bubble = []
times_insertion = []

for size in sizes:
    random_list = generate_random_list(size, -10000, 10000)
    
    # Timing bubble sort
    start_time = time.time()
    bubble_sort(random_list.copy())
    end_time = time.time()
    times_bubble.append(end_time - start_time)
    
    # Timing insertion sort
    start_time = time.time()
    insertion_sort(random_list.copy())
    end_time = time.time()
    times_insertion.append(end_time - start_time)

# Plotting the results
plt.figure(figsize=(10, 5))
plt.plot(sizes, times_bubble, label=' ', marker='o')
plt.plot(sizes, times_insertion, label=' ', marker='o')
plt.xlabel('  ')
plt.ylabel(' ()')
plt.title('     ')
plt.legend()
plt.grid(True)
plt.show()
